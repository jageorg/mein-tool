  <button id="go">Umkehren</button>


  <pre id="out"></pre>


  <script>

    document.querySelector('#go').addEventListener('click', () => {

      const s = document.querySelector('#txt').value || '';

      document.querySelector('#out').textContent = s.split('').reverse().join('');

    });
